<form action="Quiz04Result.jsp" method="post">
전화번호:
<select name = 'areaCode'  >
 <option value="02">  서울 = 02   </option>
 <option value="032"> 경기 = 032   </option>
 <option value="033"> 강원도 = 033  </option>
</select> 



<input type="number" name="fistNum" oninput="handleOnInput(this, 4)">
<input type="number" name="secNum" oninput="handleOnInput(this, 4)">
<input type="submit">
</form>
<script>
var obj = document.getElementById('areaCode');
var aa = obj.options[obj.selectedIndex].value;



//number type의 최대 길이수 제한함수
function handleOnInput(el, maxlength) {
	  if(el.value.length > maxlength)  {
	    el.value 
	      = el.value.substr(0, maxlength);
	  }
	}
	
	
	
</script>




</body>
