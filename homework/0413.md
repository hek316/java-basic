package day16.homework;
     // 1. 생성자 (여러분 마음대로 여러 개 만들기)

	// 2. getters (형식에 맞게)

	// 3. setters (재료를 넣으면 검열을 거쳐 올바른 데이터만 들어감)
	// 0) name : 이름
	// 1) kr, en, ma : 0 점 이상 100점 이하만 저장 가능, 그 외 0점
	// 2) avg : (인자값 받지 않고, kr, en, ma만 가지고 계산되도록)
	// 3) pass : 60점 이상이면 true
	// 4) grade : ABCDF 중 1개로
class Student {
	private String name;
	private int kr, en, ma;
	private double avg;
	private boolean pass;
	private String grade;

	Student() {
	}

	Student(String name, int kr, int en, int ma) {
		this.name = name;
		this.kr = kr;
		this.en = en;
		this.ma = ma;
		setAvg();
		setGrade();
		isPass();

	}

	public String getInfo() {
		return "이름 " + name + "\n kr " +kr + "\n en " +en+ "\n ma " +ma
				+"\n avg " +avg +"\n pass ?  " +pass +"\n grade ?  " +grade;

	}

	public void setNumbers(String name ,int kr, int en, int ma) {
		if (kr < 0 || kr > 100 || en < 0 || en > 100 || ma < 0 || ma > 100 ) {
			System.out.println("범위를 초과하였습니다");
			return;
		}
		this.name = name;
		this.kr = kr;
		this.en = en;
		this.ma = ma;
		setAvg();
		isPass();
		setGrade();

	}

	public void setAvg() {
		avg = (kr + ma + en) / 3.0;
	}

	public void isPass() {
		pass = (avg >= 60);
	}

	public void setGrade() {
		if (avg >= 90) {
			grade = "A";
        } else if (avg >= 80) {
			grade = "B";
		} else if (avg >= 70) {
			grade = "C";
		} else if (avg >= 60) {
			grade = "D";
		} else if (avg >= 80) {
			grade = "E";
		} else {
			grade = "F";
		}

	}

	
}
public class Test01 {
	public static void main(String[] args) {
		Student a1 = new Student("홍길동" , 100 , 90, 90);
		System.out.println(a1.getInfo());
		
		Student a2 = new Student();
		a2.setNumbers( "니니즈" ,99, 50, 100);
		System.out.println(a2.getInfo());
		
		Student a3 = new Student();
		a3.setNumbers( "단무지" ,99, 100, 100);
		System.out.println(a3.getInfo());
		
	}

}
