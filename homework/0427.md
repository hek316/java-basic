ackage day25.homework;

import java.io.FileInputStream;

import java.io.FileOutputStream;





class Timer extends Thread{
	int i = 0;
	public void run() {
	while( !isInterrupted() ) {
		try {
			System.out.println(++i+ "초");
			Thread.sleep(1000);
		} catch (InterruptedException e) {
			e.printStackTrace();
		}
	}
	}
	
}

	public static void main(String[] args) {
